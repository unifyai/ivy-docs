def _rot_mat_to_xyx_euler(rot_mat):
    # BS x 1
    euler_angles_1 = ivy.acos(rot_mat[..., 0, 0:1])

    gimbal_validity = ivy.abs(rot_mat[..., 0, 2:3]) > GIMBAL_TOL

    r23 = rot_mat[..., 1, 2:3]
    r22 = rot_mat[..., 1, 1:2]
    gimbal_euler_angles_0 = ivy.atan2(-r23, r22)
    gimbal_euler_angles_2 = ivy.zeros_like(gimbal_euler_angles_0)

    # BS x 3
    gimbal_euler_angles = ivy.concat([gimbal_euler_angles_0, euler_angles_1, gimbal_euler_angles_2], axis=-1)

    # BS x 1
    r12 = rot_mat[..., 0, 1:2]
    r13 = rot_mat[..., 0, 2:3]
    r21 = rot_mat[..., 1, 0:1]
    r31 = rot_mat[..., 2, 0:1]
    normal_euler_angles_0 = ivy.atan2(r12, r13)
    normal_euler_angles_2 = ivy.atan2(r21, -r31)

    # BS x 3
    normal_euler_angles = ivy.concat([normal_euler_angles_0, euler_angles_1, normal_euler_angles_2], axis=-1)

    return ivy.where(gimbal_validity, normal_euler_angles, gimbal_euler_angles)

[docs]def rot_mat_to_euler(rot_mat, convention='zyx'):
    """Convert rotation matrix :math:`\mathbf{R}\in\mathbb{R}^{33}` to :math:`zyx` Euler angles
    :math:`\mathbf{}_{xyz} = [_z, _y, _x]`.\n
    `[reference] <https://github.com/alisterburt/eulerangles/blob/master/eulerangles/conversions.py>`_

    Parameters
    ----------
    rot_mat
        Rotation matrix *[batch_shape,3,3]*
    convention
        The axes for euler rotation, in order of L.H.S. matrix multiplication. (Default value = 'zyx')

    Returns
    -------
    ret
        zyx Euler angles *[batch_shape,3]*

    """
    try:
        return ROT_MAT_TO_EULER_DICT[convention](rot_mat)
    except KeyError:
        raise Exception('convention must one of: {}'.format(VALID_EULER_CONVENTIONS))




[docs]def quaternion_to_euler(quaternion, convention='zyx'):
    """Convert quaternion :math:`\mathbf{q} = [q_i, q_j, q_k, q_r]` to :math:`zyx` Euler angles
    :math:`\mathbf{}_{xyz} = [_z, _y, _x]`.\n
    `[reference] <https://github.com/alisterburt/eulerangles/blob/master/eulerangles/conversions.py>`_

    Parameters
    ----------
    quaternion
        Input quaternion *[batch_shape,4]*
    convention
        The axes for euler rotation, in order of L.H.S. matrix multiplication. (Default value = 'zyx')

    Returns
    -------
    ret
        zyx Euler angles *[batch_shape,3]*

    """

    # BS x 3
    return rot_mat_to_euler(ivy_rot_mat.quaternion_to_rot_mat(quaternion), convention)

[docs]def get_random_euler(batch_shape=None):
    """Generate random :math:`zyx` Euler angles :math:`\mathbf{}_{xyz} = [_z, _y, _x]`.

    Parameters
    ----------
    batch_shape
        Shape of batch. Shape of [1] is assumed if None. (Default value = None)
